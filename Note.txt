sudo useradd sftp-user

sudo groupadd sftp

sudo usermod -G sftp sftp-user

sudo chown sftp-user:sftp /data

Match User sftp-user
        ChrootDirectory /data/%u
        ForceCommand internal-sftp
        AllowTcpForwarding no
